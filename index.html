1.2 Wordcloud Keywords

Show the code 
ERPI$Wordcloud <- 
   EMPP_2 %>% filter(Laboratoire=="ERPI") %>% 
   pull(AuthKeywords) %>% 
   tolower() %>% 
   strsplit( split=c(";")) %>% 
   unlist() %>% 
   str_trim( side = c("left")) %>% 
   as_tibble() %>% 
   group_by(value) %>% 
   drop_na() %>% 
   tally(sort = TRUE, name = "Frequency") %>% 
   wordcloud2(shape = 'circle', size=0.3 )
